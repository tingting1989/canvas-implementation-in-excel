// 设置单元格值 sheet.data.setCell(row, col, value); // 加载数组数据 sheet.data.loadData([ ["姓名", "年龄", "工资"], ["张三", 30, 25000] ]);
// 8层样式权重体系 sheet.styles.setDefaultStyle({ fontSize: 12, fontFamily: "Arial" }); sheet.styles.setCellStyle(0, 0, { bold: true, backgroundColor: "#FFEB3B" }); // 解析完整样式（应用所有层级） const fullStyle = sheet.styles.resolveStyle(0, 0);
// 合并 A1:D1 标题行 sheet.merges.mergeCells(0, 0, 0, 3); // 查询合并状态 if (sheet.merges.isMergedCell(0, 0)) { const info = sheet.merges.getMerge(0, 0); console.log(info); } // 获取所有合并区域 const allMerges = sheet.merges.getAllMerges();

// 添加条件格式：工资 > 30000 显示红色 sheet.styles.addConditionalRule({ range: { topRow: 1, topCol: 2, bottomRow: 10, bottomCol: 2 }, condition: (value) => value > 30000, style: { backgroundColor: "#FFCDD2", color: "#C62828", bold: true } });
// 设置简单列头 sheet.meta.colHeaders = ["姓名", "年龄", "工资"]; // 设置嵌套表头（多层表头） sheet.meta.nestedHeaders = [ [{ label: "基本信息", colspan: 2 }, { label: "薪资", colspan: 2 }], ["姓名", "年龄", "基本工资", "奖金"] ];
// 配置列类型 sheet.meta.applyColumnsConfig([ { col: 3, field: "salary", type: "currency", options: { symbol: "¥", decimals: 2 } }, { col: 4, field: "dept", type: "select", options: ["技术部", "市场部"] } ]); // 格式化显示 const formatted = sheet.meta.formatCellValue(3, 25000); // 返回: "¥25,000.00"
