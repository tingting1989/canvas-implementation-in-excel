📖 使用说明（UMD 方式）

<!-- 1. 引入 CanvasSheet UMD 库 -->
<script src="../dist/canvas-sheet.umd.js"></script>

// 2. 使用全局变量 CanvasSheet
const wb = new CanvasSheet.Workbook(document.getElementById('app'), {
    hooks: {
        ['onImportProgress'] (progress) {
            console.log(\`${progress.percent}%`);
        },
        
        ['onImportComplete'] (result) {
            console.log('成功!', result);
        }
    }
});

// 3. 查找插件并使用
const plugin = wb.getPlugin('importFile');

// 4. 导入文件
await plugin.importFromFile(file);
